This is a classic known-plaintext attack.
Let's create a script to find the key!

import sys

# extend alphabet if necessary
alphabet = "abcdefghijklmnopqrstuvwyxzABCDEFGHIJKLMNOPQRSTUVWYXZ"

def decode_ciphertext(ciphertext, key):
    key_pos = 0
    plaintext = ""
    for x in ciphertext:
        new_char = chr((ord(x) - ord(key[key_pos])) % 255)
        plaintext += new_char
        key_pos += 1
        key_pos = key_pos % len(key)
    return plaintext

def get_key(p1, c1):
    i = 0
    key = ""
    while i < len(p1):
        found_match = False
        for x in alphabet:
            new_char = chr((ord(p1[i]) + ord(x)) % 255)
            if new_char == c1[i]:
                print("Found match, current key is " + key)
                found_match = True
                i += 1
                key += x
        if found_match == False:
            print("Couldn't find match for char at index " + str(i))
            return 0
    return key

def main(argv):
    p1 = open(argv[0], "r").read()
    c1 = open(argv[1], "r").read()
    c2 = open(argv[2], "r").read()

    print("\nKnown plaintext: \n" + p1)
    print("Known ciphertext: \n" + c1)
    print("Ciphertext to decode: \n" + c2)

    key = get_key(p1, c1)
    if key == 0:
        print("Unable to calculate key!")
    else:
        p2 = decode_ciphertext(c2, key)
        print("Original plaintext: " + p2)

if __name__ == "__main__":
    if len(sys.argv) < 3:
        print("Please specify the following arguments:")
        print("[0] known plaintext file")
        print("[1] known ciphertext file")
        print("[2] ciphertext to decode")
    else:
        main(sys.argv[1:])

Now that we have robert's password we can SSH into the machine, get the user flag and move on to root.

Getting root

We start by doing some enumeration with LinEnum:

$ ./LinEnum.sh...
User robert may run the following commands on obscure:
(ALL) NOPASSWD: /usr/bin/python3 /home/robert/BetterSSH/BetterSSH.py

Let's run BetterSSH as sudo to test this:

$ sudo /usr/bin/python3 /home/robert/BetterSSH/BetterSSH.py

Authed!

After looking at BetterSSH.py we realise there's a vulnerability we can exploit: we can read the passwords that are stored on /etc/shadow for a period of time on /tmp/SSH/.
